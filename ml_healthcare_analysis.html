# Load cleaned dataset
df = pd.read_csv("cleaned_data.csv")

# Select low-cardinality, meaningful features
selected_cols = [
    "age", "gender", "blood_type", "medical_condition", 
    "date_of_admission", "discharge_date", "admission_type"
]
df = df[selected_cols]

# Convert dates to stay duration
df["date_of_admission"] = pd.to_datetime(df["date_of_admission"])
df["discharge_date"] = pd.to_datetime(df["discharge_date"])
df["stay_length"] = (df["discharge_date"] - df["date_of_admission"]).dt.days

# Drop original date columns
df.drop(columns=["date_of_admission", "discharge_date"], inplace=True)

# Define features and target
y = df["admission_type"]
X = pd.get_dummies(df.drop(columns=["admission_type"]))

import pandas as pd
import spacy
from transformers import pipeline

data = {
    "patient_feedback": [
        "The doctor was very kind and helpful.",
        "I had to wait too long for the test results.",
        "Excellent service and care!",
        "The billing process was confusing."
    ]
}

df = pd.DataFrame(data)

nlp = spacy.load("en_core_web_sm")

sentiment_pipeline = pipeline("sentiment-analysis")

sentiments = []
entities = []

for text in df["patient_feedback"]:
    sentiment_result = sentiment_pipeline(text)[0]
    sentiments.append(sentiment_result)

    doc = nlp(text)
    entities.append([(ent.text, ent.label_) for ent in doc.ents])


df["sentiment"] = sentiments
df["named_entities"] = entities

df.to_csv("nlp_results.csv", index=False)

print(df)

import pandas as pd
from prophet import Prophet
import matplotlib.pyplot as plt

df = pd.read_csv("cleaned_data.csv")
df['date_of_admission'] = pd.to_datetime(df['date_of_admission'])
df = df.sort_values('date_of_admission')
df = df[['date_of_admission', 'billing_amount']].rename(columns={'date_of_admission': 'ds', 'billing_amount': 'y'})

model = Prophet()
model.fit(df)

future = model.make_future_dataframe(periods=365)
forecast = model.predict(future)

fig1 = model.plot(forecast)
plt.title("Billing_Amount_Forecast")
plt.xlabel("date_of_admission")
plt.ylabel("billing_amount")
fig1.savefig("billing_forecast.png")
plt.show()

fig2 = model.plot_components(forecast)
fig2.savefig("forecast_components.png")
plt.show()

from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Image as RLImage
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib.pagesizes import letter

excel_path = "insight_report.xlsx"
df.to_excel(excel_path, index=False)
print(f"Excel report saved at: {excel_path}")

charts = [
    ("gender_distribution.png", "Patient Gender Distribution"),
    ("condition_chart.png", "Medical Condition Counts"),
    ("admission_type_chart.png", "Admission Type Distribution"),
    ("billing_amount_distribution.png", "Billing Amount Distribution"),
    ("top_seven_hospitals.png", "Top 7 Hospitals by Patient Count"),
    ("feature_importances.png", "Top 10 Feature Importances (ML Model)"),
    ("billing_forecast.png", "Billing Amount Forecast (Prophet)"),
    ("forecast_components.png", "Forecast Components (Prophet)")
]

ml_accuracy = 0.85  # example, replace with actual accuracy
ml_f1_score = 0.82  # example, replace with actual F1 score

pdf_path = "final_healthcare_report.pdf"

elements.append(Paragraph("Healthcare Data Insights and Forecast Report", styles['Title']))

    elements.append(Spacer(1, 24))

doc.build(elements)
